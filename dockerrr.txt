newgrp docker обновляет  членсто в группах

# создать группу docker
sudo groupadd docker

# вывести последние 10 строк из файла со списком групп
tail /etc/group

# добавить текущего пользователя в группу docker
sudo usermod -aG docker $USER

# вывести список групп пользователя itguy
groups itguy

# перезапустить службу docker
sudo systemctl restart docker

# перечитать список пользователей в группе docker
newgrp docker

# запустить контейнер hello-world
sudo docker run hello-world

# то же, но без указания прав root-пользователя
docker run hello-world

# запустить в фоновом режиме контейнер с документацией (на порту 80)

Другие примеры с docker exec
# Получить интерактивную оболочку в контейнере
docker exec -it my-container /bin/bash

# Выполнить однократную команду
docker exec my-container ls -la

# Проверить процессы в контейнере
docker exec my-container ps aux

# Посмотреть логи приложения
docker exec my-container tail -f /var/log/app.log
